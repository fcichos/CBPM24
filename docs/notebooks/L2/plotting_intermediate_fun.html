Exercise 1: Customizing Line Plots#

Objective: Enhance a line plot with custom styles, annotations, and shading.

Instructions:

  1. Generate a sine wave and a cosine wave using numpy.

  2. Plot both waves on the same figure with different line styles and colors.

  3. Add a legend to distinguish between the sine and cosine waves.

  4. Annotate the local maxima and minima using plt.annotate().

  5. Use plt.fill_between() to shade the area between the waves.

Instructions:

  1. Generate three different datasets using numpy (e.g., linear, quadratic, and logarithmic relationships).

  2. Create a 1x3 subplot layout with shared y-axis.

  3. Plot each dataset on a separate subplot and link their y-axes.

Exercise 3: Interactive Plotting with Widgets#

Instructions:

  1. Use ipywidgets to create a dropdown menu that lets the user select between different types of plots (e.g., line, scatter, bar).

  2. Depending on the selection, update the plot dynamically to display the chosen plot type.

  3. Include interactive widgets for customizing plot parameters like color and marker style.

import matplotlib.pyplot as plt
from matplotlib.widgets import Slider, Button

# Initial plot
fig, ax = plt.subplots()
plt.subplots_adjust(bottom=0.25)
t = range(0, 100)
s = [i**2 for i in t]
l, = plt.plot(t, s, lw=2)

# Slider for adjusting the power of the polynomial
axcolor = 'lightgoldenrodyellow'
axfreq = plt.axes([0.25, 0.1, 0.65, 0.03], facecolor=axcolor)
slider = Slider(axfreq, 'Power', 1.0, 3.0, valinit=2)

# Update function for the slider
def update(val):
    power = slider.val
    l.set_ydata([i**power for i in t])
    fig.canvas.draw_idle()

slider.on_changed(update)

# Reset button
resetax = plt.axes([0.8, 0.025, 0.1, 0.04])
button = Button(resetax, 'Reset', color=axcolor, hovercolor='0.975')

def reset(event):
    slider.reset()
button.on_clicked(reset)

plt.show()

import numpy as np
import matplotlib.pyplot as plt
import time

# Function to generate a random dataset and plot description
def generate_random_plot():
    dataset_type = np.random.choice(['line', 'scatter', 'bar'])
    x = np.linspace(0, 10, 100)
    if dataset_type == 'line':
        y = np.sin(x) + np.random.normal(size=x.size)
        plt.plot(x, y)
        plt.title("Line Plot")
    elif dataset_type == 'scatter':
        y = np.cos(x) + np.random.normal(size=x.size)
        plt.scatter(x, y)
        plt.title("Scatter Plot")
    elif dataset_type == 'bar':
        y = np.abs(np.random.normal(size=10))
        plt.bar(range(10), y)
        plt.title("Bar Chart")
    plt.show()

# Start the race
start_time = time.time()
generate_random_plot()
end_time = time.time()
print(f"Time taken: {end_time - start_time} seconds")

import matplotlib.pyplot as plt
import numpy as np

# Generate synthetic stock data
np.random.seed(0)
dates = np.arange('2020-01-01', '2020-02-01', dtype='datetime64[D]')
prices = np.random.normal(loc=100, scale=10, size=len(dates))
prices = np.cumprod(1 + prices / 100) * 100  # Cumulative product to simulate price

# Plot the time series
plt.plot(dates, prices, label='Stock Price')

# Highlight periods of consecutive increase
for i in range(5, len(prices)):
    if all(prices[j] > prices[j-1] for j in range(i-5, i)):
        plt.axvspan(dates[i-5], dates[i], color='green', alpha=0.3)

plt.xlabel('Date')
plt.ylabel('Closing Price')
plt.title('Stock Price Time Series with Consecutive Increases Highlighted')
plt.legend()
plt.show()
